#include <bits/stdc++.h>

using i64 = long long;

int main(){
	std::ios::sync_with_stdio(false);
	std::cin.tie(nullptr);

	int n, m;
	std::cin >> n >> m;
	
	std::vector<int> p(12, 0);
	for (int i = 0; i < m; i++) {
		int t;
		std::cin >> t;
		p[t] = 1;
	}
	
	int i;
	for (i = n; ; i++) {
		bool ok = true;
		 
		int t = i;
		while (t && ok) {
			int mod = t % 10;
			t /= 10;
			if (p[mod]) {
				ok = false;
			}
		}
		
		if (ok) break;
	}
	
	std::cout << i << "\n";

	return 0;
}
