class Solution {
public:
    int dp[100010];
    int sum[100010];
    int a[100010];
    int n;
    int ans = -1e9;
    int dfs(int id) {
        int &ret = dp[id];
        if (ret != 0x3f3f3f3f) return ret;
        // for (int i = 1; i < id; i ++ ) {
        //     ret = max(ret, dfs(i) + sum[id] - sum[i]);
        // }
        ret = sum[id] - sum[id - 1];
        int d = dfs(id - 1);
        if (d > 0) ret += d;
        ans = max(ans, ret);
        return ret;
    }
    int maxSubArray(vector<int>& nums) {
        n = nums.size();
        memset(dp, 0x3f, (n + 2) * sizeof(int));
        memset(sum, 0, (n + 2) * sizeof(int));
        for (int i = 1; i <= n; i ++ )
            a[i] = nums[i - 1];
        for (int i = 1; i <= n; i ++ )
            sum[i] = sum[i - 1] + a[i];
        
        ans = dp[1] = a[1];
        dfs(n);
        // for (int i = 1; i <= n; i ++ )
        //     cout << dp[i] << " ";
        return ans;
    }
};
